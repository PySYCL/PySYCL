48 /// \brief Function to check if the input value is a power of two.
49 /// \param[in] N The input value.
50 /// \return If the function is a power of two.
51 bool is_power_of_two(int N) {
52  if (N <= 0) return false;
53  return (N & (N - 1)) == 0;
54 }

56 ///////////////////////////////////////////////////////////////////////
57 /// \brief 1-dimensional fast fourier transform function kernel.
58 /// This implementation adapts the CUDA FFT approach discussed in
59 /// the GitHub repository: [roguh/cuda-fft](https://github.com/roguh/cuda-fft).
60 /// \param[in] A The input vector for fft.
61 /// \param[in] result The result vector for fft.
62 /// \param[in] Q The sycl queue.
63 template<typename Input_T, typename Output_T, typename Int_T>
64 auto cooley_turkey(Input_T* data,
65  Output_T* result_device,
66  Int_T N,
67  Int_T logn,
68  Int_T wg_size,
69  sycl::queue Q) {
70  Q.submit([&](sycl::handler& h){
71  const size_t global_size = ((N/2 + wg_size - 1)/wg_size)*wg_size;
72 
73  sycl::range<1> global{global_size};
74  sycl::range<1> local{wg_size};
75 
76  h.parallel_for(sycl::nd_range<1>(global, local), [=](sycl::nd_item<1> it){
77  const Int_T i = it.get_global_id(0);
78 
79  Int_T rev;
80 
81  rev = bit_reverse(2*i);
82  rev = rev >> (32 - logn);
83  result_device[2*i] = data[rev];
84 
85  rev = bit_reverse(2*i + 1);
86  rev = rev >> (32 - logn);
87  result_device[2*i + 1] = data[rev];
88 
89  it.barrier(sycl::access::fence_space::local_space);
90 
91  for (Int_T s = 1; s <= logn; ++s) {
92  Int_T mh = 1 << (s - 1);
93  Int_T k = (i / mh) * (1 << s);
94  Int_T j = i % mh;
95  Int_T kj = k + j;
96 
97  auto a = result_device[kj];
98  auto b = result_device[kj + mh];
99 
100  float theta = -PI * j / mh;
101  Output_T twiddle(std::cos(theta), std::sin(theta));
102  auto b_twiddle = twiddle * b;
103 
104  result_device[kj] = a + b_twiddle;
105  result_device[kj + mh] = a - b_twiddle;
106 
107  it.barrier(sycl::access::fence_space::local_space);
108  }
109  });
110  }).wait();
111 }

113 ///////////////////////////////////////////////////////////////////////
114 /// \brief 1-dimensional fast fourier transform function (cooley-turkey).
115 /// This implementation adapts the CUDA FFT approach discussed in
116 /// the GitHub repository: [roguh/cuda-fft](https://github.com/roguh/cuda-fft).
117 /// \param[in] A The input vector for fft.
118 /// \return The result of the fft
119 template <typename Vector_type>
120 auto cooley_turkey_algorithm(Vector_type &A) {
121  using Scalar_T = typename Vector_type::Scalar_T;
122  using Int_T = uint32_t;
123 
124  const Int_T N = A.get_size();
125  const Int_T logn = std::log2(N);
126 
127  const Int_T wg_size = A.dev().get_max_workgroup_size();
128 
129  std::vector<std::complex<Scalar_T>> result_host(N);
130 
131  std::complex<Scalar_T>* result_device = sycl::malloc_device<std::complex<Scalar_T>>(N, A.dev().get_queue());
132 
133  Scalar_T* data = A.get_device_data_ptr();
134  sycl::queue& Q = A.dev().get_queue();
135 
136  cooley_turkey(data, result_device, N, logn, wg_size, Q);
137 
138  A.dev().get_queue().memcpy(result_host.data(), result_device, N*sizeof(std::complex<Scalar_T>)).wait();
139  return result_host;
140 }

142 ///////////////////////////////////////////////////////////////////////
143 /// \brief 1-dimensional fast fourier transform function (chirp-z).
144 /// Reference for the Chirp Z-transform implementation, followed
145 /// [Chirp Z-transform](https://en.wikipedia.org/wiki/Chirp_Z-transform).
146 /// \param[in] A The input vector for fft.
147 /// \return The result of the fft
148 template <typename Vector_type>
149 auto chirpz(Vector_type &A) {
150  using Scalar_T = typename Vector_type::Scalar_T;
151  using Int_T = uint32_t;
152 
153  const Int_T N = A.get_size();
154 
155  // Length for zero padding
156  Int_T M = 1;
157  while(M < 2*N - 1) M *= 2;
158 
159  const Int_T wg_size = A.dev().get_max_workgroup_size();
160 
161  std::vector<std::complex<Scalar_T>> result_host(N);
162 
163  Scalar_T* data = A.get_device_data_ptr();
164  sycl::queue& Q = A.dev().get_queue();
165 
166  std::complex<Scalar_T>* a_n = sycl::malloc_device<std::complex<Scalar_T>>(M, Q);
167  std::complex<Scalar_T>* b_n = sycl::malloc_device<std::complex<Scalar_T>>(M, Q);
168  std::complex<Scalar_T>* a_n_star = sycl::malloc_device<std::complex<Scalar_T>>(M, Q);
169  std::complex<Scalar_T>* b_n_star = sycl::malloc_device<std::complex<Scalar_T>>(M, Q);
170 
171  std::complex<Scalar_T>* result_device = sycl::malloc_device<std::complex<Scalar_T>>(N, Q);
172 
173  // Sequences a and b
174  Q.submit([&](sycl::handler& h){
175  const size_t global_size = ((M + wg_size - 1)/wg_size)*wg_size;
176 
177  sycl::range<1> global{global_size};
178  sycl::range<1> local{wg_size};
179 
180  h.parallel_for(sycl::nd_range<1>(global, local), [=](sycl::nd_item<1> it){
181  const Int_T i = it.get_global_id(0);
182 
183  if(i < N) {
184  const auto theta = PI*i*i/N;
185  a_n[i] = data[i]*std::complex<Scalar_T>(cos(-theta), sin(-theta));
186  b_n[i] = std::complex<Scalar_T>(cos(theta), sin(theta));
187  }else if(i >= N && i < M - N + 1) {
188  a_n[i] = std::complex<Scalar_T>(0.0, 0.0);
189  b_n[i] = std::complex<Scalar_T>(0.0, 0.0);
190  }else {
191  const auto theta = PI*(M - i)*(M - i)/N;
192  a_n[i] = std::complex<Scalar_T>(0.0, 0.0);
193  b_n[i] = std::complex<Scalar_T>(cos(theta), sin(theta));
194  }
195  });
196  }).wait();
197 
198  const Int_T logm = std::log2(M);
199  cooley_turkey(a_n, a_n_star, M, logm, wg_size, Q);
200  cooley_turkey(b_n, b_n_star, M, logm, wg_size, Q);
201 
202  Q.submit([&](sycl::handler& h){
203  const size_t global_size = ((M + wg_size - 1)/wg_size)*wg_size;
204 
205  sycl::range<1> global{global_size};
206  sycl::range<1> local{wg_size};
207 
208  h.parallel_for(sycl::nd_range<1>(global, local), [=](sycl::nd_item<1> it){
209  const Int_T i = it.get_global_id(0);
210 
211  a_n[i] = a_n_star[i]*b_n_star[i];
212  b_n[i] = b_n_star[i];
213 
214  a_n_star[i] = std::complex<Scalar_T>(0.0, 0.0);
215  b_n_star[i] = std::complex<Scalar_T>(0.0, 0.0);
216  });
217  }).wait();
218 
219  // inverse fft a
220  Q.submit([&](sycl::handler& h){
221  const size_t global_size = ((M + wg_size - 1)/wg_size)*wg_size;
222 
223  sycl::range<1> global{global_size};
224  sycl::range<1> local{wg_size};
225 
226  h.parallel_for(sycl::nd_range<1>(global, local), [=](sycl::nd_item<1> it){
227  const Int_T i = it.get_global_id(0);
228 
229  a_n[i] = conj(a_n[i]);
230  });
231  }).wait();
232 
233  cooley_turkey(a_n, a_n_star, M, logm, wg_size, Q);
234 
235  Q.submit([&](sycl::handler& h){
236  const size_t global_size = ((M + wg_size - 1)/wg_size)*wg_size;
237 
238  sycl::range<1> global{global_size};
239  sycl::range<1> local{wg_size};
240 
241  h.parallel_for(sycl::nd_range<1>(global, local), [=](sycl::nd_item<1> it){
242  const Int_T i = it.get_global_id(0);
243 
244  a_n[i] = conj(a_n_star[i])/static_cast<Scalar_T>(M);;
245  });
246  }).wait();
247 
248  Q.submit([&](sycl::handler& h){
249  const size_t global_size = ((N + wg_size - 1)/wg_size)*wg_size;
250 
251  sycl::range<1> global{global_size};
252  sycl::range<1> local{wg_size};
253 
254  h.parallel_for(sycl::nd_range<1>(global, local), [=](sycl::nd_item<1> it){
255  const Int_T i = it.get_global_id(0);
256 
257  const auto theta = -PI*i*i/N;
258  result_device[i] = a_n[i]*std::complex<Scalar_T>(cos(theta), sin(theta));
259  });
260  }).wait();
261 
262  Q.memcpy(result_host.data(), result_device, N*sizeof(std::complex<Scalar_T>)).wait();
263 
264  sycl::free(a_n, Q);
265  sycl::free(b_n, Q);
266  sycl::free(a_n_star, Q);
267  sycl::free(b_n_star, Q);
268  sycl::free(result_device, Q);
269 
270  return result_host;
271 }

274 /// \brief 1-dimensional fast fourier transform function.
275 /// \param[in] A The input vector for fft.
276 /// \return The result of the fft
277 template <typename Vector_type>
278 auto fft1d(Vector_type &A) {
279  const auto N = A.get_size();
280 
281  if(is_power_of_two(N)) return cooley_turkey_algorithm(A);
282  else return chirpz(A);
283 }
